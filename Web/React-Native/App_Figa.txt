import { StatusBar } from 'expo-status-bar';
import React, { useState, useEffect } from 'react';
import { Button, Image, StyleSheet, Text, View } from 'react-native';
import { Audio } from 'expo-av';

export default function App() {
  const [sound, setSound] = useState(null);

  async function playSound() {
    console.log('Loading Sound');

    const { sound } = await Audio.Sound.createAsync({
      uri: 'https://limbuscompany.wiki.gg/images/LCB_Sinner_Ishmael_Morning.wav?91f3e1'
    });

    setSound(sound);

    console.log('Playing Sound');
    await sound.playAsync();
  }

  // Cleanup audio on unmount
  useEffect(() => {
    return sound
      ? () => {
          console.log('Unloading Sound');
          sound.unloadAsync();
        }
      : undefined;
  }, [sound]);

  return (
    <View style={styles.container}>
      <Button title="Play Sound" onPress={playSound} />


      <Image
        style={{ height: 500, width: 500 }}
        source={{
          uri: 'https://static.wikia.nocookie.net/limbuscompany/images/0/07/Ishmaelart2.jpg/revision/latest/scale-to-width-down/1000?cb=20230107124055'
        }}
      />

      {/* <StatusBar style="auto" /> */}
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: '#ff801e',
    alignItems: 'center',
    justifyContent: 'center',
  },
});
