import { StatusBar } from 'expo-status-bar';
import { useState } from 'react';
import { Button, FlatList, Image, Modal, StyleSheet, Text, TextInput, View } from 'react-native';

export default function App() {
  const [messaggio, setMessaggio] = useState("Ciao");
  const [visible, setVisible] = useState(false);
  const [nome, setNome] = useState("");
  const [openModal, setOpenModal] = useState(false);
  const[numero, setNumero] = useState(0);
  const[value1, setValue1] = useState(0);
  const[value2, setValue2] = useState(0);

  const dati = [
    { id: "1", nome: "Mario" },
    { id: "2", nome: "Rob" },
    { id: "3", nome: "Gino" },
  ];

  return (
    <View style={styles.container}>
      <Text>{nome}</Text>
      {visible && <Text>{messaggio}</Text>}

      <TextInput
        placeholder="Inserisci testo"
        onChangeText={setNome}
        style={styles.inputText}
        value={nome}
      />

      <Image
        style={{ height: 300, width: 300, marginVertical: 20 }}
        source={{
          uri: 'https://static.wikia.nocookie.net/limbuscompany/images/0/07/Ishmaelart2.jpg/revision/latest/scale-to-width-down/1000?cb=20230107124055'
        }}
      />

      <Button
        title="Cambia testo"
        onPress={() => setMessaggio("Ho premuto il pulsante")}
      />
      <Button
        title={visible ? "Nascondi" : "Visualizza"}
        onPress={() => setVisible(!visible)}
      />

      <Text> {numero} </Text>

      <Button title = 'Incrementa' onPress={() => setNumero(numero + 1)} />
      <Button title = 'Decrementa' onPress={() => setNumero(numero - 1)} />

      <TextInput inputMode='numeric' placeholder='Inserisci il primo numero: ' onChangeText={setValue1} value = {value1} />
      <TextInput inputMode='numeric' placeholder='Inserisci il secondo numero: ' onChangeText={setValue2} value = {value2} />
      <Text> La somma equivale a: {Number(value1) + Number(value2)} </Text>


      <View style={styles.containerList}>
        <FlatList
          data={dati}
          renderItem={({ item }) => <Text>{item.nome}</Text>}
          keyExtractor={(item) => item.id}
        />
      </View>

      <Button title="Apri Modal" onPress={() => setOpenModal(true)} />

      <Modal visible={openModal} animationType="slide">
        <View style={styles.modalContainer}>
          <Text>Sono una modal</Text>
          <Button title="Chiudi" onPress={() => setOpenModal(false)} />
        </View>
      </Modal>
      
      <StatusBar style="auto" />
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: "#ff801e",
    alignItems: "center",
    justifyContent: "center",
    paddingHorizontal: 20,
    paddingVertical: 50
  },
  containerList: {
    width: '100%',
    alignItems: "center",
    justifyContent: "center",
    marginVertical: 20
  },
  inputText: {
    borderWidth: 1,
    borderColor: "#000",
    padding: 10,
    width: '100%',
    marginVertical: 10
  },
  modalContainer: {
    flex: 1,
    alignItems: "center",
    justifyContent: "center",
    backgroundColor: "#fff"
  }
});
